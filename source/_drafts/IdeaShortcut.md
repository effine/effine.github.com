title: IDEA快捷方式
tags:
---

$ command + D 		# 复制当前行
$ command + X		# 删除当前行
$ command + H		# 显示类结构(继承)图

$ command + B 		# 查看方法定义
$ command +	option + B 		# 查看方法的实现细节

$ command +	option + 方向键(左left/右right) 		# 返回至上、下次浏览的位置

$ ctrl + tab		# 切换标签页（唤醒Swither，包含打开的文件及快捷选项入口）
$ ctrl + option + H 		# 查看方法调用结构图
$ command + f7 		# 查找方法的调用 find usages
$ command + ` 		# 切换打开的多个窗口
